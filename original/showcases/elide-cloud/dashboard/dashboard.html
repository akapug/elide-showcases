Quick Start

# Install Elide Cloud CLI npm install -g @elide-cloud/cli # Login to your platform elide-cloud login # Create a new application elide-cloud apps:create my-awesome-app # Deploy your application cd my-app elide-cloud deploy --app=my-awesome-app # Scale your application elide-cloud scale web=3:standard-2x --app=my-awesome-app # Add a database elide-cloud addons:create postgres:standard --app=my-awesome-app # View logs elide-cloud logs --app=my-awesome-app --tail=100
